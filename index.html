window.addEventListener('scroll', () => { const sections = document.querySelectorAll('.column'); const fadeStart = 400; // Starting point for fade const fadeEnd = 600; // Ending point for fade const delayFactor = 150; // Delay between each section's animation sections.forEach((section, index) => { const scrollY = window.scrollY; const delayOffset = index * delayFactor; // Add delay for each section const start = fadeStart + delayOffset; const end = fadeEnd + delayOffset; if (scrollY > start && scrollY < end) { const opacity = 1 - (scrollY - start) / (end - start); section.style.opacity = opacity; section.style.transform = `translateY(${(1 - opacity) * 20}px)`; // Optional: Adds subtle travel effect } else if (scrollY <= start) { section.style.opacity = 1; section.style.transform = 'translateY(0)'; } else { section.style.opacity = 0; section.style.transform = 'translateY(20px)'; } }); });
